# 2. Current Order Book Snapshot

HTTP Request

    # Request
    GET /b2b/nomics/rank/orders/snapshot

request Paramters

name Type Status description
market string Mandatory market ID from the /markets

Response description

name Type Status description
code code Mandatory Status code Note:200 means success and other means failure
msg String Mandatory Prompt information
timestamp Integer Mandatory the timestamp this snapshot was created in RFC3339 in UTC
bids decimal Mandatory An array containing 2 elements. The offer price and quantity fyor each bid order
asks decimal Mandatory An array containing 2 elements. The ask price and quantity for each ask order
avgResTimeMs Long Mandatory avg response time.ms
invoke boolean Mandatory true invoke history 'false' is not invoke

# 3. Candles

HTTP Request

                    
    # Request
    GET /b2b/nomics/rank/candles

request description

name Type Status description
market string Mandatory market ID from the /markets
interval string Mandatory he interval of the OHLCV candles. Valid values are 1d, 1h, 1d

response description

name Type Status description
code code Mandatory Status code Note:200 means success and other means failure
msg String Mandatory Prompt information
open string Mandatory open price of the asset in the quote currency as a string parseable to a positive number
close string Mandatory close price of the asset in the quote currency as a string parseable to a positive number
low string Mandatory lowest price of the asset in the quote currency as a string parseable to a positive number
high string Mandatory highest price of the asset in the quote currency as a string parseable to a positive number
volume string Mandatory volume of the asset in the base currency as a string parseable to a positive number. Only one of volume and volume_quote are required but both are encouraged.
volume_quote string Mandatory volume of the asset in the quote currency as a string parseable to a positive number. Only one of volume and volume_quote are required but both are encouraged.
timestamp string Mandatory timestamp of the start of the candle in RFC3339 aligned to candle size in UTC
avgResTimeMs Long Mandatory avg response time.ms
invoke boolean Mandatory true invoke history 'false' is not invoke 
    # Response

        {
            "msg": "success",
            "code": 200,
            "data": [{
                    "close": 23263.5,
                    "high": 23263.5,
                    "low": 23258.5,
                    "open": 23260.5,
                    "timestamp": "2023-02-08T10:54:00.000Z",
                    "volume": 38.0,
                    "volume_quote": 884013.0000
                }],
             "avgResTimeMs": "33",
                 "invoke": false
        }

# 5. Historical Executed Trades

HTTP Request

                    
    # Request
    GET /b2b/nomics/rank/trades

request description

name Type Status description
market string Mandatory market ID from the /markets
since string Mandatory limit

response description

name Type Status description
code code Mandatory Status code Note:200 means success and other means failure
msg String Mandatory Prompt information
id string Mandatory A string ID for the trade that is unique within the scope of the market
timestamp string Mandatory Timestamp of the trade in RFC3339 in UTC
price string Mandatory The price for one unit of the base currency expressed in the quote currency as a string that is parseable to a positive number.
amount string Mandatory The amount of the base currency that was traded as a string that is parseable to a positive number. Only one of amount and amount_quote are required but both are encouraged.
amount_quote string Mandatory The amount of the quote currency that was traded as a string that is parseable to a positive number. Only one of amount and amount_quote are required but both are encouraged.
side string Mandatory The direction of the trade [buy, sell]
avgResTimeMs Long Mandatory avg response time.ms
invoke boolean Mandatory true invoke history 'false' is not invoke

                        "code": 200,
                        "data": [{

                            "amount_quote": 124123,

                            "price": "39751.5",
                            "side": "sell",
                            "timestamp": "2022-03-16T10:04:13.000Z"
                        }],
